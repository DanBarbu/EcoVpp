How to wire this to a real backend

    
The mock generator lives in assets/app.js (function startDemo). Replace it with the real fetch like so:


// inside assets/app.js
async function loadShares() {
  const r = await fetch('/api/shares/latest', { headers: { Accept: 'application/json' } });
  const rows = await r.json();
  // ...render rows into #shares-body
}
const ws = new WebSocket(`${location.protocol === 'https:' ? 'wss' : 'ws'}://${location.host}/ws`);
ws.onmessage = (e) => {
  const { price, signal, limit_pct } = JSON.parse(e.data);
  document.getElementById('stat-price').textContent = '€' + price.toFixed(3) + '/kWh';
  document.getElementById('stat-limit').textContent = limit_pct + '%';
};

    
The repo's React dashboard at dashboard/ already speaks the /api + /ws contract — use this static page as a marketing demo and the React app for operators.
